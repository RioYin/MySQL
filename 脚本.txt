Create database duel


Create table 决斗者
(
 姓名 char(20) not null unique,
 年龄 int default 18,
 性别 char(2) check(性别 in('男' ,'女') ) ,
 Primary key(姓名)
)

Create table 决斗盘
(
 设备名称 char(20) not null unique,
 类型 char(20),
 Primary key(设备名称)
)

Create table 所属
(
 设备名称 char(20) not null unique,
 姓名 char(20),
 Primary key(设备名称),
 Foreign key(设备名称) references 决斗盘(设备名称),
 Foreign key(姓名) references 决斗者(姓名)
)

Create table 卡组
(
 卡组名称 char(20) not null unique,
 卡组核心 char(20) not null unique,
 Primary key(卡组名称,卡组核心)
)

Create table 持有
(
 姓名 char(20) not null,
 卡组名称 char(20) not null,
 卡组核心 char(20) not null,
 Primary key(姓名,卡组名称,卡组核心) ,
 Foreign key(卡组名称) references 卡组(卡组名称) ,
 Foreign key(卡组核心) references 卡组(卡组核心) ,
 Foreign key(姓名) references 决斗者(姓名)
)

Create table 卡片
(
 名字 char(20) not null unique,
 颜色 char(20),
 Primary key(名字)
)

Create table 包含
(
 卡组名称 char(20) not null,
 卡组核心 char(20) not null,
 名字 char(20) not null,
 Primary key(卡组名称,卡组核心,名字) ,
 Foreign key(卡组名称) references 卡组(卡组名称) ,
 Foreign key(卡组核心) references 卡组(卡组核心) ,
 Foreign key(名字) references 卡片(名字)
)

Create table 拥有
(
 姓名 char(20) not null,
 名字 char(20) not null,
 Primary key(姓名,名字) ,
 Foreign key(姓名) references 决斗者(姓名) ,
 Foreign key(名字) references 卡片(名字)
)

Create table 陷阱卡
(
 名字 char(20) not null unique,
 种类 char(20),
 颜色 char(20),
 Primary key(名字)
)

Create table 怪兽卡
(
 名字 char(20) not null unique,
 种类 char(20),
 颜色 char(20),
 攻击力 int default 0,
 守备力 int default 0,
 Primary key(名字)
)

Create table 魔法卡
(
 名字 char(20) not null unique,
 种类 char(20),
 颜色 char(20),
 Primary key(名字)
)
